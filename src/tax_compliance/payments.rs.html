1use crate::{tax_engine, Balance, PaymentReceipt, TaxRecord, Timestamp};

3pub(crate) fn apply_payment(
4    mut record: TaxRecord,
5    amount: Balance,
6    payment_reference: [u8; 32],
7    now: Timestamp,
8) -> (TaxRecord, PaymentReceipt) {
9    record.paid_amount = record.paid_amount.saturating_add(amount);
10    record.last_payment_at = now;
11    record.payment_reference = payment_reference;
12    record.status = tax_engine::resolve_status(record, now);
13    let receipt = PaymentReceipt {
14        property_id: record.property_id,
15        jurisdiction_code: record.jurisdiction_code,
16        reporting_period: record.reporting_period,
17        payment_reference,
18        amount_paid: amount,
19        outstanding_balance: outstanding_tax(&record),
20        settled_at: now,
21    };
22    (record, receipt)
23}

25pub(crate) fn outstanding_tax(record: &TaxRecord) -> Balance {
