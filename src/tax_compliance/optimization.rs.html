1use crate::{
2    payments, Balance, JurisdictionProfile, OptimizationPlan, PropertyAssessment, TaxRecord,
3    TaxRule, Timestamp,

6pub(crate) fn recommend_plan(
7    rule: TaxRule,
8    profile: Option<JurisdictionProfile>,
9    assessment: PropertyAssessment,
10    record: Option<TaxRecord>,
11    now: Timestamp,
12) -> OptimizationPlan {
13    let outstanding = record
14        .as_ref()
15        .map(payments::outstanding_tax)
16        .unwrap_or_default();
17    let review_exemption = assessment.exemption_override < (assessment.assessed_value / 20);
18    let estimated_discount = profile
19        .filter(|item| {
20            now <= assessment
21                .last_assessed_at
22                .saturating_add(item.optimization_window)
23        })
24        .map(|item| {
25            assessment
26                .assessed_value
27                .saturating_mul(item.early_payment_discount_basis_points as Balance)
28                / 10_000
29        })
30        .unwrap_or(0);
31    let estimated_savings = estimated_discount
32        .saturating_add(outstanding.saturating_mul(rule.penalty_basis_points as Balance) / 10_000);
33
34    OptimizationPlan {
35        estimated_savings,
36        recommended_installments: if outstanding > 0 { 2 } else { 1 },
37        should_prepay: profile
38            .map(|item| item.early_payment_discount_basis_points > 0)
39            .unwrap_or(false),
40        review_exemption,
41        supporting_reference: profile.map(|item| item.authority_hash).unwrap_or([0u8; 32]),
42    }
43}
