12/// Error types for the Property Valuation Oracle
13#[derive(Debug, Clone, PartialEq, Eq, scale::Encode, scale::Decode)]
14#[cfg_attr(feature = "std", derive(scale_info::TypeInfo))]
15pub enum OracleError {
16    /// Property not found in the oracle system
17    PropertyNotFound,
18    /// Insufficient oracle sources available
19    InsufficientSources,
20    /// Valuation data is invalid or out of range
21    InvalidValuation,
22    /// Caller is not authorized to perform this operation
23    Unauthorized,
24    /// Oracle source does not exist
25    OracleSourceNotFound,
26    /// Invalid parameters provided
27    InvalidParameters,
28    /// Error from external price feed
29    PriceFeedError,
30    /// Price alert not found
31    AlertNotFound,
32    /// Oracle source has insufficient reputation
33    InsufficientReputation,
34    /// Oracle source already registered
35    SourceAlreadyExists,
36    /// Valuation request is still pending
37    RequestPending,
38    /// Input batch exceeds the configured maximum size
39    BatchSizeExceeded,
40}
41
42impl core::fmt::Display for OracleError {
43    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
44        match self {
45            OracleError::PropertyNotFound => write!(f, "Property not found in the oracle system"),
46            OracleError::InsufficientSources => write!(f, "Insufficient oracle sources available"),
47            OracleError::InvalidValuation => write!(f, "Valuation data is invalid or out of range"),
48            OracleError::Unauthorized => {
49                write!(f, "Caller is not authorized to perform this operation")
50            }
51            OracleError::OracleSourceNotFound => write!(f, "Oracle source does not exist"),
52            OracleError::InvalidParameters => write!(f, "Invalid parameters provided"),
53            OracleError::PriceFeedError => write!(f, "Error from external price feed"),
54            OracleError::AlertNotFound => write!(f, "Price alert not found"),
55            OracleError::InsufficientReputation => {
56                write!(f, "Oracle source has insufficient reputation")
57            }
58            OracleError::SourceAlreadyExists => write!(f, "Oracle source already registered"),
59            OracleError::RequestPending => write!(f, "Valuation request is still pending"),
60            OracleError::BatchSizeExceeded => write!(f, "Batch size exceeds maximum allowed"),
61        }
62    }
63}

65impl ContractError for OracleError {

68            OracleError::PropertyNotFound => oracle_codes::ORACLE_PROPERTY_NOT_FOUND,
69            OracleError::InsufficientSources => oracle_codes::ORACLE_INSUFFICIENT_SOURCES,
70            OracleError::InvalidValuation => oracle_codes::ORACLE_INVALID_VALUATION,
71            OracleError::Unauthorized => oracle_codes::ORACLE_UNAUTHORIZED,
72            OracleError::OracleSourceNotFound => oracle_codes::ORACLE_SOURCE_NOT_FOUND,
73            OracleError::InvalidParameters => oracle_codes::ORACLE_INVALID_PARAMETERS,
74            OracleError::PriceFeedError => oracle_codes::ORACLE_PRICE_FEED_ERROR,
75            OracleError::AlertNotFound => oracle_codes::ORACLE_ALERT_NOT_FOUND,
76            OracleError::InsufficientReputation => oracle_codes::ORACLE_INSUFFICIENT_REPUTATION,
77            OracleError::SourceAlreadyExists => oracle_codes::ORACLE_SOURCE_ALREADY_EXISTS,
78            OracleError::RequestPending => oracle_codes::ORACLE_REQUEST_PENDING,
79            OracleError::BatchSizeExceeded => 4012,
80        }
81    }
82
83    fn error_description(&self) -> &'static str {
84        match self {
85            OracleError::PropertyNotFound => {
86                "The requested property does not exist in the oracle system"
87            }
88            OracleError::InsufficientSources => {
89                "Not enough oracle sources are available to provide a reliable valuation"
90            }
91            OracleError::InvalidValuation => {
92                "The valuation data is invalid, zero, or out of acceptable range"
93            }
94            OracleError::Unauthorized => {
95                "Caller does not have permission to perform this operation"
96            }
97            OracleError::OracleSourceNotFound => "The specified oracle source does not exist",
98            OracleError::InvalidParameters => "One or more function parameters are invalid",
99            OracleError::PriceFeedError => "Failed to retrieve data from external price feed",
100            OracleError::AlertNotFound => "The requested price alert does not exist",
101            OracleError::InsufficientReputation => {
102                "Oracle source reputation is below required threshold"
103            }
104            OracleError::SourceAlreadyExists => {
105                "An oracle source with this identifier already exists"
106            }
107            OracleError::RequestPending => {
108                "A valuation request for this property is already pending"
109            }
110            OracleError::BatchSizeExceeded => {
111                "The number of items in the batch exceeds the configured maximum"
112            }
113        }
114    }

116    fn error_category(&self) -> ErrorCategory {
117        ErrorCategory::Oracle
118    }
