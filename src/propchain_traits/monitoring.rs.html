105/// Errors that can be returned by the monitoring contract.
106#[derive(Debug, Clone, Copy, PartialEq, Eq, Encode, Decode)]
107#[cfg_attr(feature = "std", derive(TypeInfo))]
108pub enum MonitoringError {
109    Unauthorized,
110    ContractPaused,
111    InvalidThreshold,
112    SubscriberLimitReached,
113    SubscriberNotFound,
114}
115
116impl fmt::Display for MonitoringError {
117    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
118        match self {
119            MonitoringError::Unauthorized => write!(f, "Caller is not authorized"),
120            MonitoringError::ContractPaused => write!(f, "Monitoring contract is paused"),
121            MonitoringError::InvalidThreshold => write!(f, "Alert threshold value is invalid"),
122            MonitoringError::SubscriberLimitReached => {
123                write!(f, "Maximum subscriber limit reached")
124            }
125            MonitoringError::SubscriberNotFound => write!(f, "Subscriber not found"),
126        }
127    }
128}
129
130impl ContractError for MonitoringError {
131    fn error_code(&self) -> u32 {
132        match self {
133            MonitoringError::Unauthorized => monitoring_codes::MONITORING_UNAUTHORIZED,
134            MonitoringError::ContractPaused => monitoring_codes::MONITORING_CONTRACT_PAUSED,
135            MonitoringError::InvalidThreshold => monitoring_codes::MONITORING_INVALID_THRESHOLD,
136            MonitoringError::SubscriberLimitReached => {
137                monitoring_codes::MONITORING_SUBSCRIBER_LIMIT_REACHED
138            }
139            MonitoringError::SubscriberNotFound => {
140                monitoring_codes::MONITORING_SUBSCRIBER_NOT_FOUND
141            }
142        }
143    }
144
145    fn error_description(&self) -> &'static str {
146        match self {
147            MonitoringError::Unauthorized => "Caller does not have monitoring permissions",
148            MonitoringError::ContractPaused => "Monitoring contract is currently paused",
149            MonitoringError::InvalidThreshold => {
150                "Threshold value must be between 0 and 10 000 bips"
151            }
152            MonitoringError::SubscriberLimitReached => {
153                "Cannot add more subscribers, maximum limit reached"
154            }
155            MonitoringError::SubscriberNotFound => "The subscriber account is not registered",
156        }
157    }
158
159    fn error_category(&self) -> ErrorCategory {
160        ErrorCategory::Monitoring
161    }
162}

164/// Cross-contract interface for the monitoring system.
165#[ink::trait_definition]
166pub trait MonitoringSystem {
167    /// Record a single operation outcome. Callable by admin or authorized reporters.
168    #[ink(message)]
169    fn record_operation(
170        &mut self,
171        operation: OperationType,
172        success: bool,
173    ) -> Result<(), MonitoringError>;
174
175    /// Return accumulated metrics for a specific operation type.
176    #[ink(message)]
177    fn get_performance_metrics(&self, operation: OperationType) -> PerformanceMetrics;
178
179    /// Return metrics for all known operation types.
180    #[ink(message)]
181    fn get_all_metrics(&self) -> Vec<PerformanceMetrics>;
182
183    /// Compute and return a live health-check result based on current metrics.
184    #[ink(message)]
185    fn health_check(&self) -> HealthCheckResult;
186
187    /// Return the currently stored health status (admin-controlled).
188    #[ink(message)]
189    fn get_system_status(&self) -> HealthStatus;
190
191    /// Persist a point-in-time snapshot of aggregate metrics (circular buffer).
192    #[ink(message)]
193    fn take_metrics_snapshot(&mut self) -> Result<(), MonitoringError>;
194
195    /// Retrieve a previously stored snapshot by its buffer slot index.
196    #[ink(message)]
197    fn get_metrics_snapshot(&self, slot: u64) -> Option<MetricsSnapshot>;
198}
